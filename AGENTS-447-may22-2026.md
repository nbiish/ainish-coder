```typescript
import * as crypto from 'crypto';

async function encryptDataAtRest(plaintext: Uint8Array, associatedData: Uint8Array) {
  // 1. Receiver key pair generation
  const receiverKeys = ml_kem768.keygen();
  
  // 2. Sender encapsulates the key
  const { cipherText: kemCipherText, sharedSecret } = ml_kem768.encapsulate(receiverKeys.publicKey);
  
  // 3. Import shared secret as an AES-GCM Key
  const webcrypto = crypto.webcrypto;
  const aesKey = await webcrypto.subtle.importKey(
    'raw',
    sharedSecret,
    { name: 'AES-GCM' },
    false,
    ['encrypt', 'decrypt']
  );
  
  // 4. Encrypt using AES-256-GCM
  const nonce = webcrypto.getRandomValues(new Uint8Array(12)); // 96-bit unique nonce
  const ciphertextAes = await webcrypto.subtle.encrypt(
    { name: 'AES-GCM', iv: nonce, additionalData: associatedData },
    aesKey,
    plaintext
  );
  
  return { kemCipherText, nonce, ciphertextAes: new Uint8Array(ciphertextAes), secretKey: receiverKeys.secretKey };
}
```

### Scenario C: Integrity & Hash Validation
For any local file handling or network payload ingestion, data integrity must be verified via hashing (using SHA3-256 or SHA-256) or digital signatures (ML-DSA-65) to protect against tampered payloads. Files must be read in chunks (maximum 64KB) to prevent resource exhaustion (CWE-400).

#### Python
```python
import hashlib
from oqs import Signature

# Local chunked file integrity check
def verify_file_sha3(filepath: str, expected_hex: str) -> bool:
    hasher = hashlib.sha3_256()
    with open(filepath, 'rb') as f:
        # Read in 64KB chunks to prevent memory bloat (CWE-400)
        for chunk in iter(lambda: f.read(65536), b''):
            hasher.update(chunk)
    return hasher.hexdigest() == expected_hex

# Network payload integrity and origin validation using ML-DSA-65 signatures
def verify_network_payload(payload: bytes, signature: bytes, public_key: bytes) -> bool:
    with Signature("ML-DSA-65") as verifier:
        return verifier.verify(payload, signature, public_key)
```

#### TypeScript
```typescript
import * as fs from 'fs';
import * as crypto from 'crypto';
import { ml_dsa65 } from '@noble/post-quantum/ml-dsa.js';

// Local chunked file verification
function verifyFileSHA3(filepath: string, expectedHex: string): Promise<boolean> {
  return new Promise((resolve, reject) => {
    const hash = crypto.createHash('sha3-256');
    const stream = fs.createReadStream(filepath, { highWaterMark: 65536 }); // 64KB chunks
    
    stream.on('data', (chunk) => hash.update(chunk));
    stream.on('end', () => resolve(hash.digest('hex') === expectedHex));
    stream.on('error', (err) => reject(err));
  });
}

// Network payload signature verification
function verifyPayload(payload: Uint8Array, signature: Uint8Array, publicKey: Uint8Array): boolean {
  // Noble-curves signature: verify(signature, message, publicKey)
  return ml_dsa65.verify(signature, payload, publicKey);
}
```

### Server-Side Request Forgery Mitigation (SSRF)
```python
import ipaddress
import socket
from urllib.parse import urlparse

BLOCKED_RANGES = [
    ipaddress.ip_network("10.0.0.0/8"),
    ipaddress.ip_network("172.16.0.0/12"),
    ipaddress.ip_network("192.168.0.0/16"),
    ipaddress.ip_network("127.0.0.0/8"),
    ipaddress.ip_network("169.254.0.0/16"),
    ipaddress.ip_network("::1/128"),
    ipaddress.ip_network("fc00::/7"),
]

def validate_url(url: str, allowed_domains: set[str]) -> str:
    parsed = urlparse(url)
    if parsed.scheme != "https":
        raise ValueError("HTTPS protocol required")
    if parsed.hostname not in allowed_domains:
        raise ValueError(f"Unauthorized destination domain: {parsed.hostname}")
    
    # Check IP resolved from hostname
    ip = ipaddress.ip_address(socket.getaddrinfo(parsed.hostname, None)[0][4][0])
    for blocked in BLOCKED_RANGES:
        if ip in blocked:
            raise ValueError(f"SSRF threat blocked: Resolved target in private range {ip}")
    return url
```
